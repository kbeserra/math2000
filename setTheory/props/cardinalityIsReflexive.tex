\begin{prop}
\label{prop:cardinalityIsReflexive}
  For any set $A$, $A$ has the same cardinality as itself.
\end{prop}
\begin{proof}
  The identity map $\id_A$ is trivially a bijection from $A$ to $A$ which see that $A$ has the same cardinality of $A$.
\end{proof}

\begin{exmp}
\label{exmp:setOperations}
  Let $A=\set{a,c,e,g}$ and $B=\set{d,e,f,g}$.\\
  Then
  \begin{align*}
    A\cup B &= \set{ a,c,d,e,g,f}\,, \\
    A\cap B &= \set{ e, g}\,, \\
    A\setminus B &= \set{ a, c, }\,, \\
    B\setminus A &= \set{ d, f }\,.
  \end{align*}

\end{exmp}

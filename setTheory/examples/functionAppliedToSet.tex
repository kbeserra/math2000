\begin{exmp}
\label{exmp:functionAppliedToSet}
  Let $X=\set{1,2,3,4}$ and $Y = \set{a,b,c,d,e}$.
  Define $f:X\to Y$ as \[ f(1)=b,\,f(2)=a,\,f(3)=d,\,f(4)=b\,.\]
  Compute the following
  \begin{enumerate}
    \item $f(\set{1,4})$
    \item $f(X)$
    \item $f^{-1}(\set{a,b})$
    \item $f^{-1}(\set{c,e})$
  \end{enumerate}
\end{exmp}

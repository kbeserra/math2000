\begin{defn}
\label{defn:mutuallyDisjointCollection}
\index{disjoint!mutually}
  $\mathcal{A}$ is said to be a \emph{mutually disjoint} collection provided $\forall A,B\in\mathcal{A}$ $A$ and $B$ are disjoint.
\end{defn}

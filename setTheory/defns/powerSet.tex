\begin{defn}
\label{defn:powerSet}
\index{power set}
  Let $A$ be a set.
  The set of all subsets of $A$, $\mathcal{P}$, is called the \emph{power set} of $A$.
  \[ \mathcal{P}(A) := \set{ B\mid B\subseteq A}\,.\]
\end{defn}

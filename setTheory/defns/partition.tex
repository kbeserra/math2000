\begin{defn}
\label{defn:partition}
\index{partition}
  \sloppy $\mathcal{A}$ is said to be a \emph{partition} of a set $X$ provided ${\bigcup_{A\in\mathcal{A}} A = X}$ and $\mathcal{A}$ is a mutually disjoint collection.
\end{defn}

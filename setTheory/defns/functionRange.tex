\begin{defn}
\label{defn:functionRange}
\index{range}
\index{codomain}
  For a given function $f:X\to Y$, we call the set \[ \set{ y\in Y \mid \exists x\in X( f(x) = y)}\] the \emph{range} of $f$, and denote this set $\ran(f)$.
  We call the set $Y$ to \emph{codomain} of $f$.
\end{defn}

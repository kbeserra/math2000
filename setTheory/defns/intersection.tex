\begin{defn}
\label{defn:intersection}
\index{intersection}
  Let $A$ and $B$ sets.
  The \emph{intersection} of $A$ and $B$, $A\cap B$ is the set which contains everything which is in $A$ and in $B$.
  \[ A\cap B := \set{ x \mid x\in A \wedge x\in B}\,.\]
\end{defn}

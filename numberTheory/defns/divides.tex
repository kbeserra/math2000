\begin{defn}
\label{defn:divide}
\index{divide}
  An integer $d\not=0$ is said to \emph{divide} an integer $n$ provided, written $d|n$, provided there exists an integer $k$ such that $n=dk$.
  We will write $a\nmid b$ to mean $\neg(a|b)$.
\end{defn}

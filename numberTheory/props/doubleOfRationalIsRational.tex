\begin{prop}
\label{prop:doubleOfRationalIsRational}
  The double of a rational is rational.
\end{prop}
\begin{proof}
  Fix $r\in\QQ$.
  As $r\in\QQ$ we have by Proposition \ref{prop:sumOfRaionalsIsRational}, the double of $r$, $2r=r+r$ is rational.
\end{proof}

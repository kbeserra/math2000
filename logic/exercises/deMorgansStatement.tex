\begin{exercise}
  Use DeMorgan's, Proposition \ref{prop:DeMorgans}, to write the negation of the following statements:
  \begin{enumerate}
    \item John is $6$ feet tall or he weighs less than $200$ pounds.
    \item The bus was late or Tom's watch was slow.
    \item $-1\leq x\leq 4$.
  \end{enumerate}
\end{exercise}

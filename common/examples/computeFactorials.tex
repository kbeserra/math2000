\begin{exmp}
\label{exmp:computeFactorials}
  Compute $n!$ for $n=0,1,2,\dots,10$.
  \begin{enumerate}
    \item $0!=1$, by definiton.
    \item $1! = 1\cdot 0 !$ by definition, thus $1!= 1\cdot 1 = 1$.
    \item $2! = 2\cdot 1!= 2$
    \item $3! = 3\cdot 2! = 3\cdot 2 = 6$.
    \item $4! = 24$
    \item $5! = 120$
    \item $6! = 720$
    \item $7! = 5040$
    \item $8! = 40320$
    \item $9! = 362880$
    \item $10!= 3628800$
  \end{enumerate}
\end{exmp}

\begin{exmp}
\label{exmp:computeTermsOfSequence}
  Compute the first four terms of the sequences:
  \begin{enumerate}
    \item $a_k = \frac{k}{k+1}$ where $k\geq 1$.\\
      \[  a_1=\frac{1}{2}\quad
          a_2=\frac{2}{3}\quad
          a_3=\frac{3}{4}\quad
          a_4=\frac{4}{5} \]
    \item $b_i = \frac{i-1}{i}$ where $i\geq 2$.\\
      \[  b_2=\frac{1}{2}\quad
          b_3=\frac{2}{3}\quad
          b_4=\frac{3}{4}\quad
          b_5=\frac{4}{5} \]
    \item $c_j = (-1)^j$ where $j\in\NN$.
      \[  c_0=1\quad
          c_1=-1\quad
          c_2=1\quad
          c_3=-1 \]
  \end{enumerate}
\end{exmp}
